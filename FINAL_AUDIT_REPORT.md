class Pagination {
  static paginate(query, page = 1, limit = 10) {
    const skip = (page - 1) * limit;
    return query.skip(skip).limit(limit);
  }

  static getPaginationData(total, page, limit) {
    return {
      currentPage: page,
      totalPages: Math.ceil(total / limit),
      totalItems: total,
